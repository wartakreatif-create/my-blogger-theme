    // Alamat dasar untuk semua postingan (Default Homepage)
    var urlHasil = '/feeds/posts/summary?alt=json&max-results=' + totalBeritaPerHal + '&start-index=' + startIndex;

    // 1. Jika URL mengandung Label
    if (path.indexOf('/search/label/') !== -1) {
        var label = path.split('/search/label/')[1].split('?')[0];
        urlHasil = '/feeds/posts/summary/-/' + label + '?alt=json&max-results=' + totalBeritaPerHal + '&start-index=' + startIndex;
    } 
    // 2. Jika URL mengandung Kata Kunci Pencarian (Search)
    else if (cari.indexOf('q=') !== -1) {
        var query = new URLSearchParams(cari).get('q');
        urlHasil = '/feeds/posts/summary?alt=json&q=' + encodeURIComponent(query) + '&max-results=' + totalBeritaPerHal + '&start-index=' + startIndex;

    return urlHasil;
}

function muatBeritaGrid3(page) {
    halamanSekarang = page;
    var startIndex = ((page - 1) * totalBeritaPerHal) + 1;
    var targetArea = document.getElementById('grid3-render-area');
    var feedUrl = dapatkanUrlFeedCerdas(startIndex);
    
    targetArea.style.opacity = '0.5';
    
    fetch(feedUrl)
    .then(res => res.json())
    .then(json => {
        let html = '';
        if (json.feed.entry && json.feed.entry.length > 0) {
            json.feed.entry.forEach(post => {
                let title = post.title.$t;
                let link = post.link.find(l => l.rel === 'alternate').href;
                let tag = post.category ? post.category[0].term : 'News';
                let thumb = post.media$thumbnail ? post.media$thumbnail.url.replace('s72-c', 's400') : 'https://placehold.jp';
                
                let pubDate = new Date(post.published.$t);
                let diff = Math.floor((new Date() - pubDate) / (1000 * 60 * 60));
                let timeStr = diff < 1 ? "Baru saja" : diff + " jam lalu";
                if(diff > 24) timeStr = Math.floor(diff/24) + " hari lalu";

                html += `
                    <a href="${link}" class="wc-g3-item">
                        <div class="wc-g3-img"><img src="${thumb}" alt="${title}"/></div>
                        <div class="wc-g3-content">
                            <h4 class="wc-g3-title">${title}</h4>
                            <div class="wc-g3-meta">
                                <span class="wc-g3-label">${tag}</span> • <span>${timeStr}</span>
                            </div>
                        </div>
                    </a>`;
            });
            targetArea.innerHTML = html;
            targetArea.style.opacity = '1';
            updatePagination();
            
            // Scroll ke posisi widget jika berpindah halaman
            if(page > 1) targetArea.scrollIntoView({behavior: "smooth", block: "start"});
        } else {
            targetArea.innerHTML = '<p style="text-align:center;padding:50px;color:#888;">Hasil tidak ditemukan.</p>';
            targetArea.style.opacity = '1';
        }
    })
    .catch(err => {
        console.error("Gagal memuat feed:", err);
        targetArea.style.opacity = '1';
    });
}

function updatePagination() {
    var navBox = document.getElementById('wc-pagination-box');
    var html = '';
    
    // Navigasi halaman 1, 2, 3
    for (var i = 1; i <= 3; i++) {
        var statusAktif = (i === halamanSekarang) ? 'aktif' : '';
        html += `<div class="wc-page-num ${statusAktif}" onclick="muatBeritaGrid3(${i})">${i}</div>`;
    }
    
    html += `<div class="wc-page-next" onclick="muatBeritaGrid3(${halamanSekarang + 1})"> &rsaquo; </div>`;
    navBox.innerHTML = html;
}
